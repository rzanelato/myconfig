[user]
	email = email@email.com
	username = xusername
	name = xname

[color]
	ui = auto
	branch = auto
	diff = auto
	status = auto
	showbranch = auto
	interactive = auto

[color "branch"]
	current = yellow reverse
	local = yellow
	remote = green

[color "diff"]
	meta = yellow bold
	frag = magenta bold
	old = red bold
	new = green bold
	whitespace = red reverse

[color "status"]
	added = yellow
	changed = green
	untracked = red

[core]
    whitespace=fix,-indent-with-non-tab,trailing-space,cr-at-eol
    editor = vim
    excludesFile = ~/.gitignore
	autocrlf = input
    
[alias]
aa = add .
	am = commit -am
	ca = commit --amend
	can = commit --amend --no-edit
	co = checkout
	cm = commit -m
	cp = cherry-pick
	cr = commit --amend --reset-author --no-edit
	df = diff --color-words='[^[:space:]]'
	db = branch -D
	ft = fetch --all
	lgd = log --graph --oneline --decorate=short --branches='*'
	lo = log --oneline
	lp = log -p
	nb = checkout -b
	pf = push --force-with-lease
	pl = pull
	pnb = push -u origin
	ps = push
	rb = rebase
	rba = rebase --abort
	rbc = rebase --continue
	rbi = rebase -i
	rh = reset --hard
	ss = stash
	ssp = stash pop
	sh = show --color-words='[^[:space:]]'
	st = status -sb
	up = !git fetch --all && git pull
